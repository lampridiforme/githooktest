/**
 * Test constructor
 * @constructor
 * @param {string} title
 * @param {string} author
 */
function Book(title, author) {
	this.title = title;
	this.author = author;

	/**
	 * A function for printing book information
	 * @return {string} 
	 */
	this.printBook = () => {
		let str = this.title + " written by " + this.author;
		console.log(str);
		return str;
	}
}
